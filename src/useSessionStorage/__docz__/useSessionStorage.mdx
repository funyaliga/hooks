# useSessionStorage #

sessionStorage hooks

<hr />

## API ##
```jsx
const [
  value,
  setValue,
] = useSessionStorage(
  key: string,
  initialValue?: InitialValue<T>,
);
```


### key
localStorage的key值

### initialValue
初始值


<hr />

## DEMO ##

<Playground>
  {() => {
    const [state, setState] = useSessionStorage('s-storage-key', 'value1');
    return (
      <div>
        <div>value: {state}</div>
        <button onClick={() => setState('value2')}>set value2</button>
        <button onClick={() => setState('value3')}>set value3</button>
        <button onClick={() => setState()}>clear</button>
      </div>
    )
  }}
</Playground>
